local promptConnections = {}

local function setupPrompt(prompt, enableInstant)
    if promptConnections[prompt] then
        promptConnections[prompt]:Disconnect()
        promptConnections[prompt] = nil
    end

    if enableInstant then
         
        local originalHold = prompt.HoldDuration
        local originalLOS = prompt.RequiresLineOfSight

         
        prompt.HoldDuration = 0
        prompt.RequiresLineOfSight = false

         

         
        promptConnections[prompt] = true
    else
         
        prompt.HoldDuration = 1
        prompt.RequiresLineOfSight = true
    end
end

        return
    end

    local emptyCars = {}

    for _, car in ipairs(civCars:GetChildren()) do
        if car:IsA("Model") and car:FindFirstChild("DriveSeat") then
            local driveSeat = car:FindFirstChild("DriveSeat")
            if driveSeat:IsA("VehicleSeat") and driveSeat.Occupant == nil then
                table.insert(emptyCars, car)
            end
        end
    end

    if #emptyCars == 0 then
        Library:Notify({
            Title = "No Cars Found",
            Description = "No empty cars available!",
            Time = 5
        })
        return
    end

    for _, car in ipairs(emptyCars) do
        if not car.PrimaryPart then
            car.PrimaryPart = car:FindFirstChild("PrimaryPart") or car:FindFirstChildWhichIsA("BasePart")
            if not car.PrimaryPart then
                warn("Car has no PrimaryPart!")
                continue
            end
        end

        local carSize = car:GetExtentsSize()
        local targetPos = rootPart.Position + Vector3.new(0, carSize.Y/2 + 0.1, 0)
        local targetCFrame = CFrame.new(targetPos, targetPos + rootPart.CFrame.LookVector)

        car:PivotTo(targetCFrame)

        for _, part in ipairs(car:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Anchored = true
            end
        end
        task.wait(0.1)
        for _, part in ipairs(car:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Anchored = false
            end
        end

        local posValue = car:FindFirstChild("Position") or car:FindFirstChild("LastLocation") or car:FindFirstChild("SpawnPoint")
        if posValue and posValue:IsA("Vector3Value") then
            posValue.Value = targetPos
        end

        local driveSeat = car:FindFirstChild("DriveSeat")
        if driveSeat and driveSeat:IsA("VehicleSeat") then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                task.wait(0.1)
                driveSeat:Sit(humanoid)
            end
        end

        task.wait(1.5)

        forceJumpOut()

        task.wait(0.5)
    end

    Library:Notify({
        Title = "Success",
        Description = "Stole all empty cars.",
        Time = 5
    })
end

local function bringClosestEmptyCarToPlayer()
    local character, rootPart = getCharacterAndRootPart()
    local civCars = workspace:FindFirstChild("CivCars")
    if not civCars then
        Library:Notify({
            Title = "Error",
            Description = "CivCars folder not found!",
            Time = 5
        })
        return
    end

    local closestCar = nil
    local shortestDistance = math.huge
    local playerPosition = rootPart.Position

    for _, car in ipairs(civCars:GetChildren()) do
        if car:IsA("Model") and car:FindFirstChild("DriveSeat") then
            local driveSeat = car:FindFirstChild("DriveSeat")
            if driveSeat:IsA("VehicleSeat") and driveSeat.Occupant == nil then
                local distance = (driveSeat.Position - playerPosition).Magnitude
                if distance < shortestDistance then
                    shortestDistance = distance
                    closestCar = car
                end
            end
        end
    end

    if not closestCar then
        Library:Notify({
            Title = "No Car Found",
            Description = "Could not find any empty car nearby!",
            Time = 5
        })
        return
    end

    if not closestCar.PrimaryPart then
        closestCar.PrimaryPart = closestCar:FindFirstChild("PrimaryPart") or closestCar:FindFirstChildWhichIsA("BasePart")
        if not closestCar.PrimaryPart then
            warn("Car has no PrimaryPart!")
            return
        end
    end

    local rayOrigin = rootPart.Position + Vector3.new(0, 5, 0)
    local rayDirection = Vector3.new(0, -50, 0)
    local raycastParams = RaycastParams.new()
    raycastParams.FilterDescendantsInstances = {character, closestCar}
    raycastParams.FilterType = Enum.RaycastFilterType.Blacklist

    local rayResult = workspace:Raycast(rayOrigin, rayDirection, raycastParams)
    local groundY = rootPart.Position.Y
    if rayResult then
        groundY = rayResult.Position.Y
    end

    local carSize = closestCar:GetExtentsSize()
    local forward = rootPart.CFrame.LookVector
    local targetPos = rootPart.Position + forward * (carSize.Z / 2 + 3)
    targetPos = Vector3.new(targetPos.X, groundY + carSize.Y / 2 + 0.1, targetPos.Z)

    local targetCFrame = CFrame.new(targetPos, targetPos + forward)
    closestCar:PivotTo(targetCFrame)

    for _, part in ipairs(closestCar:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Anchored = true
        end
    end

    task.wait(0.2)

    for _, part in ipairs(closestCar:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Anchored = false
        end
    end

    local posValue = closestCar:FindFirstChild("Position") or closestCar:FindFirstChild("LastLocation") or closestCar:FindFirstChild("SpawnPoint")
    if posValue and posValue:IsA("Vector3Value") then
        posValue.Value = targetPos
    end

    task.wait(0.2)

    local driveSeat = closestCar:FindFirstChild("DriveSeat")
    if driveSeat and driveSeat:IsA("VehicleSeat") then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            task.wait(0.1)
            driveSeat:Sit(humanoid)
        end
    end

    Library:Notify({
        Title = "Success",
        Description = "Nearest car brought to you!",
        Time = 5
    })
end

-- Cosmetic: change some button text where quick to detect (best-effort, non-invasive)
safe_pcall(function()
    -- search workspace, StarterGui, or Library UI for TextLabels/TextButtons and update a few known strings
    local function tryReplaceGuiStrings(root)
        if not root then return end
        for _, inst in ipairs(root:GetDescendants()) do
            if inst:IsA("TextLabel") or inst:IsA("TextButton") or inst:IsA("TextBox") then
                local txt = inst.Text
                if type(txt) == "string" then
                    -- replace a few common words
                    txt = txt:gsub("Welcome @", "Welcome, mortal @")
                    txt = txt:gsub("Welcome", "Welcome (to Halloween)")
                    txt = txt:gsub("Infinite", "Eternal")
                    txt = txt:gsub("Auto", "Ghoul-Auto")
                    txt = txt:gsub("Money", "Candy")
                    inst.Text = txt
                end
            end
        end
    end

    pcall(function() tryReplaceGuiStrings(game:GetService("Players").LocalPlayer:WaitForChild("PlayerGui")) end)
    pcall(function() tryReplaceGuiStrings(game:GetService("StarterGui")) end)
end)

-- End of Halloween Patch
