local promptConnections = {}

local function setupPrompt(prompt, enableInstant)
    if promptConnections[prompt] then
        promptConnections[prompt]:Disconnect()
        promptConnections[prompt] = nil
    end

    if enableInstant then
         
        local originalHold = prompt.HoldDuration
        local originalLOS = prompt.RequiresLineOfSight

         
        prompt.HoldDuration = 0
        prompt.RequiresLineOfSight = false

         

         
        promptConnections[prompt] = true
    else
         
        prompt.HoldDuration = 1
        prompt.RequiresLineOfSight = true
    end
end

local InstantInteractionToggle = Tab1:AddToggle("InstantInteraction", {
    Text = "Instant Interaction",
    Default = false,
    Tooltip = "Makes ProximityPrompts instant and disables line of sight",
    Callback = function(Value)
         
        for prompt, conn in pairs(promptConnections) do
            if typeof(conn) == "RBXScriptConnection" then
                conn:Disconnect()
            end
            promptConnections[prompt] = nil
        end

         
        for _, prompt in pairs(workspace:GetDescendants()) do
            if prompt:IsA("ProximityPrompt") then
                setupPrompt(prompt, Value)
            end
        end

         
        if Value then
            promptConnections["AddedConnection"] = workspace.DescendantAdded:Connect(function(desc)
                if desc:IsA("ProximityPrompt") then
                    setupPrompt(desc, true)
                end
            end)
        else
            if promptConnections["AddedConnection"] then
                promptConnections["AddedConnection"]:Disconnect()
                promptConnections["AddedConnection"] = nil
            end
        end

        notifyToggle("Instant Interaction", Value)
    end
})


local StudioMoneyNotifyConnections = {}

local StudioMoneyNotifyToggle = Tab1:AddToggle("StudioMoneyNotify", {
    Text = "Studio Money Notify",
    Default = false,
    Tooltip = "Notifies when studio money prompt is enabled/disabled",
    Callback = function(enabled)
         
        for _, conn in ipairs(StudioMoneyNotifyConnections) do
            conn:Disconnect()
        end
        StudioMoneyNotifyConnections = {}

        if not enabled then return end

        local Workspace = game:GetService("Workspace")
        local StudioPayFolder = Workspace:FindFirstChild("StudioPay")
        if not StudioPayFolder then
            warn("StudioPay folder not found in Workspace.")
            return
        end
        local MoneyFolder = StudioPayFolder:FindFirstChild("Money")
        if not MoneyFolder then
            warn("Money folder not found inside StudioPay.")
            return
        end

        local function watchPrompt(studioPayName)
            local studioPay = MoneyFolder:FindFirstChild(studioPayName)
            if not studioPay then return end
            local part = studioPay:FindFirstChild(studioPayName)
            if not part then return end
            local prompt = part:FindFirstChild("Prompt")
            if not prompt or not prompt:IsA("ProximityPrompt") then return end

            local conn = prompt:GetPropertyChangedSignal("Enabled"):Connect(function()
                if not StudioMoneyNotifyToggle.Value then return end -- toggle off? do nothing

                if prompt.Enabled then
                    Library:Notify({
                        Title = "Studio Money Restocked",
                        Description = "Money in studio restocked.",
                        Time = 5,
                        SoundId = 87082710377514
                    })
                else
                    Library:Notify({
                        Title = "Studio Money Stolen",
                        Description = "1 Money in studio just got stolen!",
                        Time = 5,
                        SoundId = 87082710377514
                    })
                end
            end)
            table.insert(StudioMoneyNotifyConnections, conn)
        end

         
        for _, name in ipairs({"StudioPay1", "StudioPay2", "StudioPay3"}) do
            watchPrompt(name)
        end

         
        local moneyConn = MoneyFolder.ChildAdded:Connect(function(child)
            if not StudioMoneyNotifyToggle.Value then return end
            if child.Name == "StudioPay1" or child.Name == "StudioPay2" or child.Name == "StudioPay3" then
                watchPrompt(child.Name)
            end
        end)
        table.insert(StudioMoneyNotifyConnections, moneyConn)
    end

    end
end

            Title = "Error",
            Description = "CivCars folder not found!",

        })
        return
    end

    local emptyCars = {}

    for _, car in ipairs(civCars:GetChildren()) do
        if car:IsA("Model") and car:FindFirstChild("DriveSeat") then
            local driveSeat = car:FindFirstChild("DriveSeat")
            if driveSeat:IsA("VehicleSeat") and driveSeat.Occupant == nil then
                table.insert(emptyCars, car)
            end
        end
    end

    if #emptyCars == 0 then
        Library:Notify({
            Title = "No Cars Found",
            Description = "No empty cars available!",
            Time = 5
        })
        return
    end

    for _, car in ipairs(emptyCars) do
        if not car.PrimaryPart then
            car.PrimaryPart = car:FindFirstChild("PrimaryPart") or car:FindFirstChildWhichIsA("BasePart")
            if not car.PrimaryPart then
                warn("Car has no PrimaryPart!")
                continue
            end
        end

        local carSize = car:GetExtentsSize()
        local targetPos = rootPart.Position + Vector3.new(0, carSize.Y/2 + 0.1, 0)
        local targetCFrame = CFrame.new(targetPos, targetPos + rootPart.CFrame.LookVector)

        car:PivotTo(targetCFrame)

        for _, part in ipairs(car:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Anchored = true
            end
        end
        task.wait(0.1)
        for _, part in ipairs(car:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Anchored = false
            end
        end

        local posValue = car:FindFirstChild("Position") or car:FindFirstChild("LastLocation") or car:FindFirstChild("SpawnPoint")
        if posValue and posValue:IsA("Vector3Value") then
            posValue.Value = targetPos
        end

        local driveSeat = car:FindFirstChild("DriveSeat")
        if driveSeat and driveSeat:IsA("VehicleSeat") then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                task.wait(0.1)
                driveSeat:Sit(humanoid)
            end
        end

        task.wait(1.5)

        forceJumpOut()

        task.wait(0.5)
    end

    Library:Notify({
        Title = "Success",
        Description = "Stole all empty cars.",

local function bringClosestEmptyCarToPlayer()
    local character, rootPart = getCharacterAndRootPart()
    local civCars = workspace:FindFirstChild("CivCars")
    if not civCars then
        Library:Notify({
            Title = "Error",
            Description = "CivCars folder not found!",
            Time = 5
        })
        return
    end

    local closestCar = nil
    local shortestDistance = math.huge
    local playerPosition = rootPart.Position

    for _, car in ipairs(civCars:GetChildren()) do
        if car:IsA("Model") and car:FindFirstChild("DriveSeat") then
            local driveSeat = car:FindFirstChild("DriveSeat")
            if driveSeat:IsA("VehicleSeat") and driveSeat.Occupant == nil then
                local distance = (driveSeat.Position - playerPosition).Magnitude
                if distance < shortestDistance then
                    shortestDistance = distance
                    closestCar = car
                end
            end
        end
    end

    if not closestCar then
        Library:Notify({
            Title = "No Car Found",
            Description = "Could not find any empty car nearby!",
            Time = 5
        })
        return
    end

    if not closestCar.PrimaryPart then
        closestCar.PrimaryPart = closestCar:FindFirstChild("PrimaryPart") or closestCar:FindFirstChildWhichIsA("BasePart")
        if not closestCar.PrimaryPart then
            warn("Car has no PrimaryPart!")
            return
        end
    end

    local rayOrigin = rootPart.Position + Vector3.new(0, 5, 0)
    local rayDirection = Vector3.new(0, -50, 0)
    local raycastParams = RaycastParams.new()
    raycastParams.FilterDescendantsInstances = {character, closestCar}
    raycastParams.FilterType = Enum.RaycastFilterType.Blacklist

    local rayResult = workspace:Raycast(rayOrigin, rayDirection, raycastParams)
    local groundY = rootPart.Position.Y
    if rayResult then
        groundY = rayResult.Position.Y
    end

    local carSize = closestCar:GetExtentsSize()
    local forward = rootPart.CFrame.LookVector
    local targetPos = rootPart.Position + forward * (carSize.Z / 2 + 3)
    targetPos = Vector3.new(targetPos.X, groundY + carSize.Y / 2 + 0.1, targetPos.Z)

    local targetCFrame = CFrame.new(targetPos, targetPos + forward)
    closestCar:PivotTo(targetCFrame)

    for _, part in ipairs(closestCar:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Anchored = true
        end
    end

    task.wait(0.2)

    for _, part in ipairs(closestCar:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Anchored = false
        end
    end

    local posValue = closestCar:FindFirstChild("Position") or closestCar:FindFirstChild("LastLocation") or closestCar:FindFirstChild("SpawnPoint")
    if posValue and posValue:IsA("Vector3Value") then
        posValue.Value = targetPos
    end

    task.wait(0.2)

    local driveSeat = closestCar:FindFirstChild("DriveSeat")
    if driveSeat and driveSeat:IsA("VehicleSeat") then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            task.wait(0.1)
            driveSeat:Sit(humanoid)
        end
    end

    Library:Notify({
        Title = "Success",
        Description = "Nearest car brought to you!",
        Time = 5
    })
end
